/**
 * @param {number[]} height
 * @return {number}
 */
var trap = function(height) {
    const n = height.length;
    if(n === 0){
        return 0
    }

    const leftMax = new Array(n).fill(0); 
    leftMax[0] = height[0];
    for(let i = 1; i < n; i++){
        leftMax[i] = Math.max(leftMax[i - 1],height[i]);//得到当前元素左边最高的
    }
    const rightMax = new Array(n).fill(0);
    rightMax[n - 1] = height[n - 1]
    for(let i = n - 2; i >= 0; i--){
        rightMax[i] = Math.max(rightMax[i + 1],height[i]);//得到当前元素右边最高的
    }
    let ans = 0;
    for(let i = 0; i < n; i++){
        ans += Math.min(leftMax[i], rightMax[i]) - height[i];//用左右两边矮的减去当前元素本身的高度
    }
    return ans;
};
