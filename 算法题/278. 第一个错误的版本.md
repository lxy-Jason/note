/**
 * @param {function} isBadVersion()
 * @return {function}
 */
var solution = function(isBadVersion) {
    /**
     * @param {integer} n Total versions
     * @return {integer} The first bad version
     */
    return function(n) {
        let left = 1, right = n
        while(left < right){
            let mid = Math.floor(left + (right - left) / 2)
            console.log(mid)
            if(isBadVersion(mid)){
                right = mid
            }
            else{
                left = mid + 1
            }
        }
        return left
    };
};
