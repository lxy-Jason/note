    const list = [];
    preorderTraversal(root,list);
    const size = list.length;
    for(let i = 1; i < size; i++){
        const prev = list[i - 1];
        const cur = list[i];
        prev.left = null;
        prev.right = cur;
    }
};

const preorderTraversal = (root,list) => {
    if(root !== null){
        list.push(root);
        preorderTraversal(root.left,list);
        preorderTraversal(root.right,list);
    }
}
