```js
const axios = require("axios");
const request = axios.create({
  baseURL: "",
  timeout: 1000,
});
let count = 0;
let timer;
request.interceptors.response.use(
  (res) => {
    count = 0
    return res;
  },
  (err) => {
    count++;
    console.log(count);
    if (count >= 10) {
      console.log("出错了");
      clearInterval(timer);
      count = 0
    }
  }
);
function polling() {
  timer = setInterval(() => {
    getData();
  }, 1000);
}
async function getData() {
  const res = await request.get("http://api.test.com")
}

polling();
```
