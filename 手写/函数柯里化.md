```js
function toCurry(func,...args) {
    return function () {
        //合并上一次缓存的参数和本次传入的参数
        args = [...args,...arguments]
        //判断参数数量是否足够
        if(args.length < func.length) {
            //如果不够,继续递归
            //每一次递归都会形成新的闭包
            //保证柯里化函数每一步调用都是独立的,互不影响
            return toCurry(func,...args)
        }
        else{
            //参数足够,执行函数
            return func.apply(null,args)
        }
    }
}
function bar(a, b, c) {
    return a + b + c;
}
console.log(toCurry(bar)(1)(2)(3));
```
