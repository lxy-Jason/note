function selectSort(arr){
  let len = arr.length  
  let minIndex;
  for(let i = 0; i < len - 1;i++){
    minIndex = i
    for(let j = i + 1; j < len; j++){
      if(arr[j] < arr[minIndex]){
        minIndex = j
      }
    }
    let temp = arr[i];
    arr[i] = arr[minIndex]
    arr[minIndex] = temp
  }
  return arr
}
console.log(selectSort([1,2,534,54,234,64,5436,453,43]));
