### 延长生命周期

```js
function outer(){
    let count = 0
    return function(){
        return count++
    }
}
let f = outer()
console.log(f());
console.log(f());
console.log(f());
```

### 私有变量

```js
function People(name){
  return {
    getName(){
      return name
    }
  }
}
const Jason = People("Jason")
console.log(Jason.name); //undefined
console.log(Jason.getName()); //Jason
```
