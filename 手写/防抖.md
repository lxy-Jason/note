### 立即执行版本

```js
function debounce(func, wait, immediate) {
      let timeout;
      return function () {
        let context = this;
        let args = arguments;
        if (timeout) {
          clearTimeout(timeout)
        }
        if (immediate) {
          let callNow = !timeout;//第一次立即执行,以后只有事件执行才会触发
          console.log(callNow);
          timeout = setTimeout(() => {
            timeout = null
          }, wait)
          if (callNow) {
            func.apply(context, args)
          }
        }
        else {
          timeout = setTimeout(function () {
            func.apply(context, args)
          }, wait)
        }
      }
    }
```

**注意这里的立即执行是每次触发都立即执行,然后在规定时间内再次触发不执行,且重新开始计时**

不是第一次触发立即执行一次的意思
